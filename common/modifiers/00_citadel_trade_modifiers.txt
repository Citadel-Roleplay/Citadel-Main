# Character Modifiers Commands:

# Give yourself and all your vassals the modifier for 10 years:
# effect this = { add_character_modifier = { modifier = citadel_construction_t1 days = -1 } every_vassal_or_below = { add_character_modifier = { modifier = citadel_construction_t1 days = -1 } } } 
# This command removes the modifier:
# effect this = { remove_character_modifier = citadel_construction_t1 every_vassal_or_below = { remove_character_modifier = citadel_construction_t1 } }

# County Modifiers Commands:

# Apply County Modifier to all holdings ONLY YOU hold, not your vassals !
# effect this = { every_held_title = { limit = { is_county = yes } add_county_modifier = { modifier = citadel_fort_t1 years = 10 } } }
# Remove County Modifier ONLY FROM YOU:
# effect this = { every_held_title = { limit = { is_county = yes } remove_county_modifier = citadel_fort_t1 } }

# Apply County Modifier to all holdings your vassals hold, YOU NEED TO APPLY IT AGAIN FOR YOURSELF !
# effect this = { every_vassal_or_below = { limit = { is_landed = yes } every_held_title = { limit = { is_county = yes } add_county_modifier = { modifier = citadel_fort_t1 years = 10 } } } }
