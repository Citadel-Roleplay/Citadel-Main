on_game_start_after_lobby = {
    on_actions = {
        agot_brightboar_on_game_start_after_lobby
		agot_brightboar_set_legacies
		agot_brightboar_easter_eggs_special
    }
}

agot_brightboar_on_game_start_after_lobby = {
	effect = {
		set_global_variable = {
            name = brightboar_enabled
            value = yes 
        }
        debug_log = brightboar_load_game_log
        set_global_variable = {
            name = BRIGHTBOAR_Ver_1_01_00
            value = yes 
        }
		
		# agot_brightboar_dragons_setup = yes
		agot_brightboar_historical_artifacts_setup = yes
		
		dynasty:dynn_Rowan = {
			add_dynasty_modifier = golden_tree_dynasty_modifier
		}
		
		character:Gardener_3 = {
			change_age = 30
		}
		
		character:Stark_12 = {
			set_father = character:Gardener_3
		}
		
		character:Lannister_29 = {
			set_mother = character:Rowan_6
		}
		
		if = {
			limit = {
				exists = character:Osgrey_55
				has_game_rule = agot_brightboar_easter_eggs_enabled
			}
			character:Osgrey_55 = {
				give_nickname = nick_the_grand_glazer
			}
		}
		
		every_living_character = {
			if = {
				limit = { 
					dynasty ?= dynasty:dynn_Rowan
					has_trait = knight
				}
                house_rowan_knight_effect = yes
			}
			if = {
				limit = { 
					dynasty ?= dynasty:dynn_Rowan
					has_trait = squire
				}
                house_rowan_knight_effect = yes
			}
			if = {
				limit = { 
					dynasty ?= dynasty:dynn_Florent
					NOT = {
						OR = {
							title:e_the_reach.holder = {
								dynasty = { has_dynasty_perk = florent_legacy_1 }
							}
							title:e_the_reach.holder = {
								dynasty = { has_dynasty_perk = gardener_legacy_1 }
							}
						}
					}
				}
				add_unpressed_claim = title:e_the_reach
				add_unpressed_claim = title:d_highgarden
				add_unpressed_claim = title:c_manderside
				add_unpressed_claim = title:c_highgarden
			}
		}
		
		every_player = {
			if = {
				limit = {
					has_trait = dynasty_legacy_customizer
				}
				trigger_event = {
					id = agot_brightboar_dynasty_legacy_customizer.0001
				}
			}
		}
	}
}
