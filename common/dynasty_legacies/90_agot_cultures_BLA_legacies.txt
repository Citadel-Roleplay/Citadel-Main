westerosi_valyrian_legacy_track = {
	is_shown = {
		OR = { 
			culture = {
				OR = {
					this = culture:westerosi_valyrian
					any_parent_culture_or_above = {
						this = culture:westerosi_valyrian
						has_cultural_pillar = heritage_valyrian
					}
				}
			}
			
		}
	}
}
