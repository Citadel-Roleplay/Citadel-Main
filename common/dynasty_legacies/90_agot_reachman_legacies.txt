reachman_legacy_track = { # Just disables the normal Reachmen Legacy for the Honeywiners, instead they get their own one.
	is_shown = {
		NOT = { 
			culture = culture:storm_marcher
		}
		OR = { 
			AND = {
				culture = {
					OR = {
						this = culture:reachman
						any_parent_culture_or_above = {
							this = culture:reachman
						}
					}
				}
				NOT = { culture = culture:honeywiner }
			}
			dynasty = { has_dynasty_perk = reachman_legacy_1 }
		}
	}
}
