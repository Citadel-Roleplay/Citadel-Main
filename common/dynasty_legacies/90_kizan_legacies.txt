# KIZAN LEGACIES

clawman_legacy_track = {
	is_shown = {
		OR = { 
			culture = {
				OR = {
					this = culture:clawman
					any_parent_culture_or_above = {
						this = culture:clawman
					}
				}
			}
			dynasty = { has_dynasty_perk = clawman_legacy_1 }
		}
	}
}
